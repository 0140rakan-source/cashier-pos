import React, { useEffect, useState } from 'react';
import { HashRouter, Routes, Route, Navigate } from 'react-router-dom';';
import { useAuthStore } from './store';
import Layout from './components/layout/Layout';
import Login from './pages/Login';
import Dashboard from './pages/Dashboard';
import Products from './pages/Products';
import Categories from './pages/Categories';
import Customers from './pages/Customers';
import Suppliers from './pages/Suppliers';
import Inventory from './pages/Inventory';
import Purchases from './pages/Purchases';
import Expenses from './pages/Expenses';
import Reports from './pages/Reports';
import Settings from './pages/Settings';
import Users from './pages/Users';
import Shifts from './pages/Shifts';
import POS from './pages/POS';
import ActivationScreen from './pages/ActivationScreen';
import Roles from './pages/Roles';
import AuditTrail from './pages/AuditTrail';

// ─── Permission Guard — redirects instead of crashing ─────────────────────────────
function Protected({ perm, children }) {
  const { can } = useAuthStore();
  if (!can(perm)) return <Navigate to="/dashboard" replace />;
  return children;
}

// ─── App Router ───────────────────────────────────────────────────────────────────
// ─── Lock Screen Component ────────────────────────────────────────────────────
function LockScreen() {
  const { unlockWithPin, user } = useAuthStore();
  const [pin, setPin] = useState('');
  const [error, setError] = useState('');
  const [trying, setTrying] = useState(false);

  const handleUnlock = async () => {
    if (!pin) return;
    setTrying(true);
    setError('');
    const ok = await unlockWithPin(pin);
    if (!ok) setError('رمز PIN غير صحيح');
    setPin('');
    setTrying(false);
  };

  return (
    <div className="h-screen flex items-center justify-center bg-gradient-to-br from-gray-900 to-gray-700">
      <div className="bg-white rounded-2xl shadow-xl p-8 w-full max-w-sm text-center">
        <div className="w-16 h-16 bg-amber-100 rounded-full flex items-center justify-center mx-auto mb-4">
          <span className="text-3xl">🔒</span>
        </div>
        <h2 className="text-xl font-bold mb-1">الشاشة مقفلة</h2>
        <p className="text-gray-500 text-sm mb-6">{user?.fullName || 'Cashier'}</p>
        <input
          type="password"
          inputMode="numeric"
          maxLength={6}
          value={pin}
          onChange={e => setPin(e.target.value.replace(/\D/g, ''))}
          onKeyDown={e => e.key === 'Enter' && handleUnlock()}
          placeholder="ادخل رمز PIN"
          className="w-full text-center text-3xl tracking-widest font-bold border rounded-xl p-4 bg-gray-50 mb-3"
          autoFocus
        />
        {error && <p className="text-red-500 text-sm mb-3">{error}</p>}
        <button
          onClick={handleUnlock}
          disabled={trying || !pin}
          className="w-full py-3 bg-brand-500 text-white rounded-xl hover:bg-brand-600 disabled:opacity-50 font-bold"
        >
          {trying ? '...' : 'فتح القفل'}
        </button>
      </div>
    </div>
  );
}

export default function App() {
  const { user, token, checkAuth, isLocked } = useAuthStore();
  const [loading, setLoading] = useState(true);

  // PRACTICAL PROTECTION: This prevents casual copying but is not uncrackable.
  // A determined user could remove the localStorage check. For stronger protection,
  // consider server-side license validation on each API call.
  const isActivated = localStorage.getItem('license_activated') === 'true';

  // On mount: if token exists, restore user session
  useEffect(() => {
    const init = async () => {
      if (token && !user) {
        await checkAuth();
      }
      setLoading(false);
    };
    init();
  }, []);

  // Show activation screen if not activated
  if (!isActivated) {
    return <ActivationScreen />;
  }

  // Show loading while restoring session
  if (loading) {
    return (
      <div className="h-screen flex items-center justify-center bg-gray-50">
        <div className="text-center">
          <div className="w-12 h-12 border-4 border-brand-500 border-t-transparent rounded-full animate-spin mx-auto mb-4" />
          <p className="text-gray-500 text-sm">جاري التحميل...</p>
        </div>
      </div>
    );
  }

  // Show lock screen if locked (session stays active)
  if (isLocked && user) {
    return <LockScreen />;
  }

  return (
    <BrowserRouter>
      <Routes>
        {/* Login */}
        <Route path="/login" element={user ? <Navigate to="/dashboard" /> : <Login />} />

        {/* Full-screen POS */}
        <Route path="/pos" element={user ? <POS /> : <Navigate to="/login" replace />} />

        {/* Full-screen Shifts */}
        <Route path="/shifts" element={user ? <Shifts /> : <Navigate to="/login" replace />} />

        {/* Layout-wrapped routes */}
        <Route path="/" element={user ? <Layout /> : <Navigate to="/login" replace />}>
          <Route index element={<Navigate to="/dashboard" replace />} />
          <Route path="dashboard" element={<Dashboard />} />
          <Route path="products" element={<Protected perm="products.view"><Products /></Protected>} />
          <Route path="categories" element={<Protected perm="categories.view"><Categories /></Protected>} />
          <Route path="customers" element={<Protected perm="customers.view"><Customers /></Protected>} />
          <Route path="suppliers" element={<Protected perm="suppliers.view"><Suppliers /></Protected>} />
          <Route path="inventory" element={<Protected perm="inventory.view"><Inventory /></Protected>} />
          <Route path="purchases" element={<Protected perm="purchases.view"><Purchases /></Protected>} />
          <Route path="expenses" element={<Protected perm="expenses.view"><Expenses /></Protected>} />
          <Route path="reports" element={<Protected perm="reports.view"><Reports /></Protected>} />
          <Route path="users" element={<Protected perm="users.view"><Users /></Protected>} />
          <Route path="settings" element={<Protected perm="settings.view"><Settings /></Protected>} />
          <Route path="roles" element={<Protected perm="roles.view"><Roles /></Protected>} />
          <Route path="audit" element={<Protected perm="settings.view"><AuditTrail /></Protected>} />
        </Route>

        {/* Fallback */}
        <Route path="*" element={<Navigate to={user ? '/dashboard' : '/login'} replace />} />
      <Routes/>
    <BrowserRouter>
  );
}
